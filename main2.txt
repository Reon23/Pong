#include <iostream>
#include "SFML\Graphics.hpp"
#include "SFML\Window.hpp"
#include "SFML\System.hpp"
using namespace sf;
int main()
{
    RenderWindow window(VideoMode(800, 800), "SFML works!");
    window.setFramerateLimit(60);

    CircleShape triangle;
    triangle.setRadius(50.f);
    triangle.setPointCount(3);

    Vertex line[] = { Vertex(Vector2f(100.f,400.f)), Vertex(Vector2f(50.f,100.f)) };

    while (window.isOpen())
    {
        float x = 0.1f, y = 0.1f;
        Event event;
        while (window.pollEvent(event))
        {
            if (event.type == Event::Closed)
                window.close();

            if (event.type == Event::KeyPressed && event.key.code == sf::Keyboard::Escape)
            {
                window.close();
            }
        }
        //Update Section
        
        if (Keyboard::isKeyPressed(Keyboard::Q))    //Keyboard Input
        {
            window.close();
        }

        if (Mouse::isButtonPressed(Mouse::Right))   //Mouse Input
        {
            window.close();
        }

        window.clear();     //Clears previous frame

        //Draw
        window.draw(line, 2, Lines);
        window.draw(triangle);
        window.display();
    }




	return 0;
}
