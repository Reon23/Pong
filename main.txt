#include <iostream>
#include "SFML\Graphics.hpp"
#include "SFML\Window.hpp"
#include "SFML\System.hpp"

int main()
{
    sf::RenderWindow window(sf::VideoMode(800, 800), "SFML works!");
    window.setFramerateLimit(69);
    
    sf::CircleShape circle(10.f);
    circle.setFillColor(sf::Color::Green);
    circle.setPosition(sf::Vector2f(0, 0));

    sf::RectangleShape rect(sf::Vector2f(50.f,50.f));
    rect.setFillColor(sf::Color::Red);
    rect.setPosition(sf::Vector2f(50, 69));
    

    while (window.isOpen())
    {
        float x = 0.1f, y = 0.1f;
        sf::Event event;
        while (window.pollEvent(event))
        {
            if (event.type == sf::Event::Closed)
                window.close();
        }
        //Update Section
        rect.move(sf::Vector2f(x, y));
        circle.move(sf::Vector2f(0.1f, 0.1f));

        window.clear();     //Clears previous frame

        //Draw
        window.draw(circle);
        window.draw(rect);
        window.display();
    }




	return 0;
}
